SwiftUI Fruit Examples

Displaying lists and navigation in SwiftUI

1) Display a List of Fruits

Problem Statement

Write SwiftUI code to display a list of fruits: ["Apple", "Banana", "Orange", "Mango"]

SwiftUI Question: 
import SwiftUI

struct ContentView: View {
    let fruits = ["Apple", "Banana", "Orange", "Mango"]
    
    var body: some View {
        
       
    }
}
SwiftUI Answer : 
import SwiftUI

struct ContentView: View {
    let fruits = ["Apple", "Banana", "Orange", "Mango"]
    
    var body: some View {
        List(fruits, id: \.self) { fruit in
            Text(fruit)
                .font(.system(size: 18, weight: .medium))
                .padding(.vertical, 8)
        }
        .navigationTitle("Fruits")
    }
}

2) Navigation to Fruit Detail View

Problem Statement

Implement navigation between a list of fruits and a detailed view for each fruit.

SwiftUI Question: 
import SwiftUI

struct ContentView: View {
    let fruits = ["Apple", "Banana", "Orange"]
    
    var body: some View {
       
    }
}

struct FruitDetailView: View {
    let fruit: String
    
    var body: some View {
        VStack(spacing: 20) {
            Text(fruit)
            Text("This is the detail view for \(fruit).")
        }
        .navigationTitle(fruit)
    }
}
SwiftUI Answer : 
import SwiftUI

struct ContentView: View {
    let fruits = ["Apple", "Banana", "Orange"]
    
    var body: some View {
        NavigationView {
            List(fruits, id:  \.self) { fruit in
                NavigationLink(destination: FruitDetailView(fruit: fruit)) {
                    Text(fruit)
                }
            }
            .navigationTitle("Fruits")
        }
    }
}

struct FruitDetailView: View {
    let fruit: String
    
    var body: some View {
        VStack(spacing: 20) {
            Text(fruit)
            Text("This is the detail view for \(fruit).")
        }
        .navigationTitle(fruit)
    }
}
